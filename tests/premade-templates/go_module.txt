package main

import (
	"os"

	"github.com/raindevteam/gorml"
	"gopkg.in/sorcix/irc.v1"
)

type Test struct { *module.Module }

func (m *Test) Greet(msg *irc.Message, args []string) {
	m.Say(msg.Args[0], "Hello there!")
}

func main() {
	m := &Test{module.NewModule("Test", "Your module's short description")}

	m.AddCommand("greet", &module.Command{
		Help: "It greets you!",
		Fun:  m.Info,
	})

	m.Register(os.Args)
}
