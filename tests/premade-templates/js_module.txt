"use strict";

const Module = require('jsrml');

class Test extends Module {
    constructor() { 
        super("Test", "An echo module");
    }

    echo(msg, args) {
        this.say(args.join(" "));
	}
}

if (require.main === module) {
    const m = new Test();

    m.addCommand("echo", {
        Help: 'Repeats its arguments',
        Fun: m.echo
    });

    m.register(process.argv);
}
